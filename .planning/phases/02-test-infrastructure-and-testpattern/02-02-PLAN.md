    **Design for configurable responses (D-02):**
    The mock port script accepts an optional command-line argument: a path to a response configuration file. This file is Erlang term format (`:erlang.term_to_binary` / `File.read! |> :erlang.binary_to_term`), containing a map of `%{cmd_byte => {status_byte, message_binary}}`. When the mock receives a command, it checks the map; if the command byte has an entry, it uses that response; otherwise it defaults to `{0, ""}` (success, empty message) per D-03.

    **test/support/mock_port_script.exs** — Standalone script run via `elixir`:
    ```elixir
    # Mock port script that speaks the Papyrus binary protocol.
    # Accepts optional arg: path to response config file (Erlang binary term).
    # Config format: %{cmd_byte => {status_byte, message_binary}}
    # Default: status=0 (ok), empty message for all commands.

    defmodule MockPortMain do
      def run do
        responses = load_responses()
        :io.setopts(:standard_io, binary: true, encoding: :latin1)
        loop(responses)
      end

      defp load_responses do
        case System.argv() do
          [path] when is_binary(path) ->
            case File.read(path) do
              {:ok, data} -> :erlang.binary_to_term(data)
              _ -> %{}
            end
          _ -> %{}
        end
      end

      defp loop(responses) do
        case IO.binread(:stdio, 5) do
          :eof -> System.halt(0)
          {:error, _} -> System.halt(0)
          <<cmd::8, len::32-big>> ->
            if len > 0 do
              case IO.binread(:stdio, len) do
                :eof -> System.halt(0)
                {:error, _} -> System.halt(0)
                _payload -> :ok
              end
            end
            {status, msg} = Map.get(responses, cmd, {0, ""})
            msg_len = byte_size(msg)
            IO.binwrite(:stdio, <<status::8, msg_len::32-big, msg::binary>>)
            loop(responses)
        end
      end
    end

    MockPortMain.run()
    ```

    **IMPORTANT implementation note on IO.binread/IO.binwrite:** Elixir's `IO.binread(:stdio, n)` may not work correctly for raw binary port I/O. The mock script may need to use `:erlang` primitives instead. If `IO.binread` doesn't work with binary stdin, use this approach:
    ```elixir
    # Open raw stdin/stdout file descriptors
    stdin = :erlang.open_port({:fd, 0, 0}, [:binary, :eof])
    # Then receive data via port messages
    ```
    The executor should test the script manually (`echo -ne '\x01\x00\x00\x00\x00' | elixir test/support/mock_port_script.exs | xxd`) and adjust the I/O approach if needed.

    **lib/papyrus/mock_port.ex** — Helper module:
    ```elixir
    defmodule Papyrus.MockPort do
      @moduledoc """
      Provides helpers for the mock port used in testing.

      The mock port is an Elixir script that speaks the Papyrus
      length-prefixed binary protocol over stdin/stdout. It responds
      with `{:ok, ""}` to every command by default.

      ## Configuring responses per-test (D-02)

      To test error paths, create a response file that maps command bytes
      to `{status, message}` tuples:

          responses = %{0x03 => {1, "hardware fault"}}  # :clear returns error
          path = Papyrus.MockPort.write_response_file(responses, test_name)
          {:ok, pid} = Papyrus.Display.start_link(
            display_module: MyDisplay,
            port_binary: Papyrus.MockPort.port_executable(path)
          )

      ## Default usage

          {:ok, pid} = Papyrus.Display.start_link(
            display_module: MyDisplay,
            port_binary: Papyrus.MockPort.port_executable()
          )
      """

      @doc """
      Return the path to the wrapper shell script (no response config).

      The mock will return `{:ok, ""}` for every command.
      """
      @spec port_executable() :: String.t()
      def port_executable do
        wrapper_path()
      end

      @doc """
      Return a `{path, args}` tuple for use with Port.open, passing the
      response file path as an argument.

      Since `Papyrus.Display` uses `Port.open({:spawn_executable, path}, ...)`
      without args support, this function returns just the wrapper path.
      The response file path is baked into a per-invocation wrapper script.
      """
      @spec port_executable(String.t()) :: String.t()
      def port_executable(response_file_path) do
        # Create a temporary wrapper script that passes the response file arg
        tmp_dir = System.tmp_dir!()
        hash = :erlang.phash2(response_file_path, 1_000_000)
        tmp_wrapper = Path.join(tmp_dir, "papyrus_mock_#{hash}.sh")

        script_path = mock_script_path()

        File.write!(tmp_wrapper, """
        #!/bin/sh
        exec elixir "#{script_path}" "#{response_file_path}"
        """)

        File.chmod!(tmp_wrapper, 0o755)
        tmp_wrapper
      end

      @doc """
      Write a response configuration file for the mock port.

      `responses` is a map of `%{cmd_byte => {status_byte, message_binary}}`.
      Command bytes: 0x01 = :init, 0x02 = :display, 0x03 = :clear, 0x04 = :sleep.

      Returns the path to the written file.

      ## Example

          responses = %{0x03 => {1, "clear failed"}}
          path = MockPort.write_response_file(responses, "my_test")
      """
      @spec write_response_file(map(), String.t()) :: String.t()
      def write_response_file(responses, label \\ "test") when is_map(responses) do
        tmp_dir = System.tmp_dir!()
        path = Path.join(tmp_dir, "papyrus_mock_responses_#{label}_#{System.unique_integer([:positive])}.bin")
        File.write!(path, :erlang.term_to_binary(responses))
        path
      end

      @doc "Map of command atoms to their byte values, for building response configs."
      @spec command_byte(atom()) :: non_neg_integer()
      def command_byte(:init), do: 0x01
      def command_byte(:display), do: 0x02
      def command_byte(:clear), do: 0x03
      def command_byte(:sleep), do: 0x04

      defp wrapper_path do
        Path.join([project_root(), "test", "support", "mock_port.sh"])
        |> Path.expand()
      end

      defp mock_script_path do
        Path.join([project_root(), "test", "support", "mock_port_script.exs"])
        |> Path.expand()
      end

      defp project_root do
        # Navigate from priv dir up to project root
        :code.priv_dir(:papyrus)
        |> to_string()
        |> Path.join("..")
        |> Path.expand()
      end
    end
    ```
